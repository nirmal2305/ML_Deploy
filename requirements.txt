import sklearn
import joblib
import pandas as pd
import flask

# Get current versions
sk_version = sklearn.__version__
joblib_version = joblib.__version__
pd_version = pd.__version__
flask_version = "3.1.3" # Pinning to a stable version known to be in Colab

# Generate correct requirements.txt format
correct_requirements = f"""flask=={flask_version}
joblib=={joblib_version}
pandas=={pd_version}
scikit-learn=={sk_version}
gunicorn"""

with open('requirements.txt', 'w') as f:
    f.write(correct_requirements.strip())

print("Corrected requirements.txt created with the following content:")
print(correct_requirements.strip())
